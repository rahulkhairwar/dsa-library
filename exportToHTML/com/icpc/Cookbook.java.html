25           static void findShortestPaths(int start) 
26           { 
27               distance = new int[nodes]; 
28               for (int i = 0; i < nodes; i++) 
29                   distance[i] = INFINITY; 
30               TreeSet<Node> set = new TreeSet<>(); 
31               distance[start] = 0; 
32               set.add(new Node(start, 0, -1)); 
33               while (set.size() > 0) 
34               { 
35                   Node curr = set.pollFirst(); 
36                   for (Edge edge : adj[curr.node]) 
37                   { 
38                       if (distance[curr.node] + edge.weight < distance[edge.to]) 
39                       { 
40                           set.remove(new Node(edge.to, distance[edge.to], -1)); 
41                           distance[edge.to] = distance[curr.node] + edge.weight; 
42                           set.add(new Node(edge.to, distance[edge.to], curr.node)); 
43                       } 
44                   } 
45               } 
46           }

317          void dfs(int node) 
318          { 
319              visited[node] = true; 
320              visitingTime[node] = low[node] = currTime++; 
321   
322              Iterator<Integer> iterator = adj[node].iterator(); 
323              boolean isArticulationPoint = false; 
324   
325              while (iterator.hasNext()) 
326              { 
327                  int curr = iterator.next(); 
328   
329                  if (!visited[curr]) 
330                  { 
331                      parent[curr] = node; 
332                      dfs(curr); 
333                      low[node] = Math.min(low[node], low[curr]); 
334   
335                      if (visitingTime[node] <= low[curr])    // condition becomes < for bridges. 
336                          isArticulationPoint = true; 
337                  } 
338                  else if (curr != parent[node]) 
339                      low[node] = Math.min(low[node], visitingTime[curr]); 
340              } 
341   
342              if (isArticulationPoint) 
343                  articulationPoints++; 
344          }

393      { 
394          int data[]; 
395          Node segmentTree[];

397          void buildTree(int node, int treeStart, int treeEnd) 
398          { 
399              if (treeStart > treeEnd) 
400                  return; 
401              if (treeStart == treeEnd) 
402              { 
403                  segmentTree[node] = new Node(data[treeStart]); 
404                  return; 
405              } 
406   
407              int mid = (treeStart + treeEnd) / 2; 
408              buildTree(2 * node, treeStart, mid); 
409              buildTree(2 * node + 1, mid + 1, treeEnd); 
410              segmentTree[node] = new Node(Math.min(segmentTree[2 * node].minimum, segmentTree[2 * node + 1].minimum)); 
411          } 
412   
413          int queryTree(int node, int treeStart, int treeEnd, int rangeStart, int rangeEnd) 
414          { 
415              Node tempNode = segmentTree[node]; 
416   
417              if (tempNode.update > 0) 
418              { 
419                  tempNode.minimum += tempNode.update; 
420   
421                  if (treeStart != treeEnd) 
422                  { 
423                      int update = tempNode.update; 
424   
425                      segmentTree[2 * node].update = update; 
426                      segmentTree[2 * node + 1].update = update; 
427                  } 
428   
429                  tempNode.update = 0; 
430              } 
431   
432              if (treeStart > treeEnd || treeStart > rangeEnd || treeEnd < rangeStart) 
433                  return Integer.MAX_VALUE; 
434   
435              if (treeStart >= rangeStart && treeEnd <= rangeEnd) 
436                  return tempNode.minimum; 
437   
438              int mid, leftChildMin, rightChildMin; 
439              mid = (treeStart + treeEnd) / 2; 
440              leftChildMin = queryTree(2 * node, treeStart, mid, rangeStart, rangeEnd); 
441              rightChildMin = queryTree(2 * node + 1, mid + 1, treeEnd, rangeStart, rangeEnd); 
442   
443              return Math.min(leftChildMin, rightChildMin); 
444          } 
445   
446          void updateTree(int node, int treeStart, int treeEnd, int rangeStart, int rangeEnd, int addValue) 
447          { 
448              Node tempNode = segmentTree[node]; 
449   
450              if (tempNode.update > 0)    // lazy 
451              { 
452                  tempNode.minimum += tempNode.update; 
453                  if (treeStart != treeEnd) 
454                  { 
455                      segmentTree[2 * node].update += tempNode.update; 
456                      segmentTree[2 * node + 1].update += tempNode.update; 
457                  } 
458                  tempNode.update = 0; 
459              } 
460              if (treeStart > treeEnd || treeStart > rangeEnd || treeEnd < rangeStart) 
461                  return; 
462              if (treeStart >= rangeStart && treeEnd <= rangeEnd) 
463              { 
464                  tempNode.minimum += addValue; 
465                  if (treeStart != treeEnd) 
466                  { 
467                      segmentTree[2 * node].update += addValue; 
468                      segmentTree[2 * node + 1].update += addValue; 
469                  } 
470                  return; 
471              } 
472   
473              int mid = (treeStart + treeEnd) / 2; 
474              updateTree(2 * node, treeStart, mid, rangeStart, rangeEnd, addValue); 
475              updateTree(2 * node + 1, mid + 1, treeEnd, rangeStart, rangeEnd, addValue); 
476              tempNode.minimum = Math.min(segmentTree[2 * node].minimum, segmentTree[2 * node + 1].minimum); 
477          }

589          { 
590              int pos = word.charAt(index) - '0'; 
591              int reqPos = pos == 1 ? 0 : 1; 
592              Node temp = node.next[pos]; 
593              Node reqTemp = node.next[reqPos]; 
594              if (index == 31) 
595              { 
596                  if (reqTemp != null && reqTemp.isWord > 0) 
597                      return 1; 
598                  return 0; 
599              } 
600              if (reqTemp != null) 
601              { 
602                  if (reqTemp.exists > 0) 
603                      return (1 << (31 - index)) + maxXor(reqTemp, word, index + 1); 
604                  if (temp != null) 
605                      return Math.max(maxXor(reqTemp, word, index + 1), maxXor(temp, word, index + 1)); 
606                  return maxXor(reqTemp, word, index + 1); 
607              } 
608              if (temp != null) 
609                  return maxXor(temp, word, index + 1); 
610              return 0;

626      { 
627          static final int INFINITY = (int) 2e9; 
628          int v, e, s, t, flow, maxFlow; 
629          Edge[] edges;

633          void solve() 
634          { 
635              v = in.nextInt();    // no. of vertices. 
636              e = in.nextInt();    // no. of edges. 
637              s = in.nextInt();    // source. 
638              t = in.nextInt();    // sink. 
639              maxFlow = 0; 
640              edges = new Edge[e << 1]; 
641              adj = new ArrayList[v]; 
642   
643              for (int i = 0; i < v; i++) 
644                  adj[i] = new ArrayList<>(); 
645              int counter = 0; 
646              for (int i = 0; i < e; i++) 
647              { 
648                  int from, to, weight; 
649                  from = in.nextInt(); 
650                  to = in.nextInt(); 
651                  weight = in.nextInt(); 
652                  // forward edge from -> to with weight = weight 
653                  edges[counter] = new Edge(to, weight); 
654                  adj[from].add(counter++); 
655                  // backward edge from <- to with weight = 0 
656                  edges[counter] = new Edge(from, 0); 
657                  adj[to].add(counter++); 
658              } 
659              while (true) 
660              { 
661                  boolean[] visited = new boolean[v]; 
662                  Queue<Integer> queue = new LinkedList<>(); 
663                  flow = 0; 
664                  par = new int[v]; 
665                  parEdge = new int[v]; 
666                  Arrays.fill(par, -1); 
667                  Arrays.fill(parEdge, -1); 
668                  visited[s] = true; 
669                  queue.add(s); 
670                  while (queue.size() > 0) 
671                  { 
672                      int curr = queue.poll(); 
673                      if (curr == t) 
674                          break;    // breaking immediately if we reach the sink. 
675                      for (int ed : adj[curr]) 
676                      { 
677                          int to = edges[ed].to; 
678                          int weight = edges[ed].weight; 
679                          if (weight > 0 && !visited[to]) 
680                          { 
681                              visited[to] = true; 
682                              queue.add(to); 
683                              par[to] = curr; 
684                              parEdge[to] = ed; 
685                          } 
686                      } 
687                  } 
688                  augment(t, INFINITY); 
689                  if (flow == 0) 
690                      break;    // no more augmenting paths can increase the flow. 
691                  maxFlow += flow; 
692              } 
693              System.out.println("The Max Flow possible in the given graph is : " + maxFlow); 
694          }

696          void augment(int node, int minEdge) 
697          { 
698              if (node == s) 
699                  flow = minEdge; 
700              else if (par[node] != -1) 
701              { 
702                  int ed = parEdge[node]; 
703                  int weight = edges[ed].weight; 
704                  augment(par[node], Math.min(minEdge, weight)); 
705                  edges[ed].weight -= flow;    // decreasing flow from the forward edge 
706                  edges[ed ^ 1].weight += flow;    // increasing flow in the backward edge 
707              }

746                  to = in.nextInt() - 1;

748                  edges[counter] = new Edge(to, weight, counter + 2);

752                  edges[counter] = new Edge(from, weight, counter - 2); 
753                  adj[to].add(counter++); 
754                  edges[counter] = new Edge(to, 0, -1); 
755                  adj[from].add(counter++); 
756              } 
757              while (true) 
758              {

761                  flow = 0;

784                          }

787                  augment(t, INFINITY);

802                  int oppEdge = edges[ed].oppEdge; 
803                  augment(par[node], Math.min(minEdge, weight));

806                  if (oppEdge >= 0) 
807                      edges[oppEdge].weight -= flow;

1575         public int read() 
1576         { 
1577             if (numChars == -1) 
1578                 throw new InputMismatchException(); 
1579  
1580             if (curChar >= numChars) 
1581             { 
1582                 curChar = 0;

1591                     return -1;

1599             int c = read(); 
1600  
1601             while (isSpaceChar(c)) 
1602                 c = read(); 
1603

1605

1607             {

1618  
1619                 res *= 10; 
1620                 res += c & 15; 
1621  
1622                 c = read(); 
1623             } while (!isSpaceChar(c)); 
1624
